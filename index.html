💻 Dev Tip of the Week

Debounce your search inputs for better UX:

// Debounce utility
const debounce = (fn, ms) => {
  let timer;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(
      () => fn(...args), ms
    );
  };
};

// Usage
input.addEventListener(
  'input',
  debounce(search, 300)
);
